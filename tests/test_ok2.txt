#include <stdio.h>
#include <stdlib.h>

float** Matrix(float** mat, int n)
{
  int i, j;
  float tmp;

  for(i = 0; i < n; i++)
  {
    for (j = i + 1; j < n; j++)
    {
      tmp = mat[i][j];
      mat[i][j] = mat[j][i];
      mat[j][i] = tmp;
    }
  }

  for (i = 0; i < n; i++)
  {
    for (j = 0; j < n / 2; j++)
    {
       tmp = mat[i][j];
       mat[i][j] = mat[i][n - 1 - j];
       mat[i][n - 1 - j] = tmp;
    }
  }
  return mat;
}
